import javax.crypto.Cipher;
import javax.crypto.KeyGenerator;
import javax.crypto.SecretKey;
import java.util.Base64;
import java.util.Scanner;
public class AES {
    public static void main(String[] args) throws Exception {
        Scanner sc = new Scanner(System.in);
        System.out.print("Enter text to encrypt: ");
        String msg= sc.nextLine();
        SecretKey key = KeyGenerator.getInstance("AES").generateKey();
        Cipher c = Cipher.getInstance("AES");
        c.init(Cipher.ENCRYPT_MODE, key);
        String encrypted = Base64.getEncoder().encodeToString(c.doFinal(msg.getBytes()));
        System.out.println("Encrypted: " + encrypted);
        c.init(Cipher.DECRYPT_MODE, key);
        String decrypted = new String(c.doFinal(Base64.getDecoder().decode(encrypted)));
        System.out.println("Decrypted: " + decrypted);
    }
}

3. Implementation of RSA based signature system
package ISLab;
import java.math.BigInteger;
import java.util.Scanner;
class RSA {
    public static void main(String args[]) {
        Scanner sc = new Scanner(System.in);
        System.out.print("Enter first prime number p: ");
        int p = sc.nextInt();
        System.out.print("Enter second prime number q: ");
        int q = sc.nextInt();
        System.out.print("Enter the message (number) to encrypt: ");
        int msg = sc.nextInt();
        int n = p * q;
        int phi = (p - 1) * (q - 1);
        int e = 0, d = 0;
        System.out.println("n = " + n);
        System.out.println("phi = " + phi);
        for (int i = 2; i < phi; i++) {
            if (gcd(i, phi) == 1) {
                e = i;
                break;
            }
        }
        System.out.println("Public key (e) = " + e);
        for (int i = 0; i < 10; i++) {
            int x = 1 + (i * phi);
            if (x % e == 0) {
                d = x / e;
                break;
            }
        }
        System.out.println("Private key (d) = " + d);
        BigInteger M = BigInteger.valueOf(msg);
        BigInteger N = BigInteger.valueOf(n);
        BigInteger E = BigInteger.valueOf(e);
        BigInteger D = BigInteger.valueOf(d);
        BigInteger C = M.modPow(E, N);
        System.out.println("Encrypted message: " + C);
        BigInteger decrypted = C.modPow(D, N);
        System.out.println("Decrypted message: " + decrypted);
    }
    static int gcd(int a, int b) {
        if (a == 0)
            return b;
        return gcd(b % a, a);
    }
}

4. Implementation of Subset sum
import java.util.*;
public class SubsetSum {
    static boolean subsetSum(int[] a, int n, int sum) {
        if (sum == 0) return true;
        if (n == 0) return false;
        return subsetSum(a, n - 1, sum) || subsetSum(a, n - 1, sum - a[n - 1]);
    }
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        System.out.print("Enter number of elements: ");
        int n = sc.nextInt();
        int a[] = new int[n];
        System.out.println("Enter array elements:");
        for (int i = 0; i < n; i++)
            a[i] = sc.nextInt();
        System.out.print("Enter the target sum: ");
        int sum = sc.nextInt();
        if (subsetSum(a, n, sum))
            System.out.println("Subset with the given sum exists.");
        else
            System.out.println("No subset with the given sum.");
    }
}

5. Authenticating the given signature using the MD5 hash algorithm.
import java.security.*;
import java.util.Scanner;
public class MD5{
    public static void main(String[] args) throws Exception {
        Scanner sc = new Scanner(System.in);
        System.out.print("Enter data: ");
        String data = sc.nextLine();
        MessageDigest md = MessageDigest.getInstance("MD5");
        byte[] hash = md.digest(data.getBytes());
        StringBuilder sb = new StringBuilder();
        for (byte b : hash) sb.append(String.format("%02x", b));
        System.out.println("Generated signature (MD5 hash): " + sb.toString());
        sc.close();
    }
}

10. Implementation of Kerberos cryptosystem
import javax.crypto.*;
import java.util.*;
public class kerberos {
    public static void main(String[] args) throws Exception {
        Scanner sc = new Scanner(System.in);
        System.out.print("Enter your ID: ");
        String id = sc.nextLine();
        System.out.print("Enter your message: ");
        String msg = sc.nextLine();
        long t = System.currentTimeMillis() + 10000;
        SecretKey key = KeyGenerator.getInstance("AES").generateKey();
        Cipher c = Cipher.getInstance("AES");
        c.init(Cipher.ENCRYPT_MODE, key);
        byte[] enc = c.doFinal((id + ":" + t + ":" + msg).getBytes());
        String ticket = Base64.getEncoder().encodeToString(enc);
        System.out.println("\n[AS → Client] TGT and session key received. Ticket valid for 10 seconds.\n");
        System.out.println("[TGS → Client] Service ticket and session key received. Ticket valid for 10 seconds.\n");
        c.init(Cipher.DECRYPT_MODE, key);
        String[] data = new String(c.doFinal(Base64.getDecoder().decode(ticket))).split(":");
        if (data[0].equals(id) && System.currentTimeMillis() <= Long.parseLong(data[1])) {
            System.out.println("[Service] User verified successfully! ID: " + data[0] + " (ticket valid)");
            System.out.println("[Service] Message received: " + data[2]);
        } else
            System.out.println("[Service] Verification failed! Ticket expired or invalid.");
    }
}

13. Message Authentication Codes.
package ISLab;
import javax.crypto.Mac;
import javax.crypto.spec.SecretKeySpec;
import java.util.*;
public class SHA256 {
    static String hmac(String msg, String key) throws Exception {
        var mac = Mac.getInstance("HmacSHA256");
        mac.init(new SecretKeySpec(key.getBytes(), "HmacSHA256"));
        return Base64.getEncoder().encodeToString(mac.doFinal(msg.getBytes()));
    }
    public static void main(String[] args) throws Exception {
        var sc = new Scanner(System.in);
        System.out.print("Enter message: ");
        var msg = sc.nextLine();
        System.out.print("Enter secret key: ");
        var k1 = sc.nextLine();
        var mac1 = hmac(msg, k1);
        System.out.println("\n[Sender] MAC Address: " + mac1);
        System.out.print("[Receiver] Enter secret key: ");
        var k2 = sc.nextLine();
        var mac2 = hmac(msg, k2);
        System.out.println("[Receiver] MAC Address: " + mac2);
        System.out.println("\nDo sender and receiver secret keys match? " + k1.equals(k2));
        System.out.println(mac1.equals(mac2) ?
                "MAC verification successful!" :
                "MAC verification FAILED!");
    }
}
